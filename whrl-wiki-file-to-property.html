Wiki word to property

(defun whrl-wiki-file-to-property (wiki-link)
  ;; Ask for wiki-word and set it as an org property <2016-02-04 Do>  
  (interactive (list
     (read-string "wiki-link: " 
                  (file-name-base                       ;; only file base name without dir
                   (read-file-name "Enter wiki file: " 
                                   (concat "~/wiki/" (org-entry-get (point) "WIKI")) ;; Suggest existing value of WIKI property 
                                   ))))
               )
  (org-entry-put (point) "WIKI"  wiki-link) ;; update value of WIKI property
  )
